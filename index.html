import seaborn as sns
import matplotlib.pyplot as plt

# Set Seaborn style
#sns.set_theme(style="darkgrid")

# Create figure
fig, ax = plt.subplots(figsize=(10, 5), facecolor="lightgrey")

# Define color palette based on rainfall difference
bar_colors = ["green" if val > 0 else "red" for val in dfrain.yoydiff]

# Plot bar chart
sns.barplot(x=dfrain.year, y=dfrain.cms, hue=dfrain.year, palette="pastel", ax=ax)

# Customize labels and title
ax.set_xlabel("Year of Rainfall", fontsize=12, fontweight="bold")
ax.set_ylabel("Rainfall in Cms", fontsize=12, fontweight="bold")
ax.set_title("All India Rainfall (Seaborn)", fontsize=14, fontweight="bold")

# Add gridlines for better readability
ax.grid(axis="y", linestyle="--", alpha=0.7)

# Annotate bars with values
for i, v in enumerate(dfrain.cms):
    ax.text(i, v + 1, f"{v:.1f}", ha="center", fontsize=10, color="black")

plt.show()

import seaborn as sns
import matplotlib.pyplot as plt

# Set Seaborn style
sns.set_style("whitegrid")

# Create subplot grid
fig, ax = plt.subplots(nrows=2, ncols=2, figsize=(12, 5))

# Panel 1: cms
sns.barplot(x="year", y="cms", data=dfrain, ax=ax[0, 0], color="green")
ax[0, 0].set_facecolor("lightgrey")
ax[0, 0].set_xticks(range(len(dfrain.year)))
ax[0, 0].set_xticklabels(dfrain.year)

# Panel 2: yoydiff with conditional colors
barcolors = ["green" if val > 0 else "red" for val in dfrain.yoydiff]
sns.barplot(x="year", y="yoydiff", data=dfrain, ax=ax[0, 1], hue=dfrain.year,palette=barcolors)
ax[0, 1].set_facecolor("lightgrey")
ax[0, 1].set_xticks(range(len(dfrain.year)))
ax[0, 1].set_xticklabels(dfrain.year)

# Panel 3: cms again
sns.barplot(x="year", y="cms", data=dfrain, ax=ax[1, 0], hue=dfrain.year,palette=barcolors)
ax[1, 0].set_facecolor("lightgrey")
ax[1, 0].set_xticks(range(len(dfrain.year)))
ax[1, 0].set_xticklabels(dfrain.year)

# Panel 4: deviation with conditional colors
barcolors = ["green" if val > 0 else "red" for val in dfrain.deviation]
sns.barplot(x="year", y="deviation", data=dfrain, ax=ax[1, 1], 
                hue=dfrain.year,palette=barcolors, legend=False)
ax[1, 1].set_facecolor("lightgrey")
ax[1, 1].set_xticks(range(len(dfrain.year)))
ax[1, 1].set_xticklabels(dfrain.year)

plt.tight_layout()
plt.show()
